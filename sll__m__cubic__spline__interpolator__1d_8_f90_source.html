136  subroutine spline_interpolate1d_disp_inplace(this, num_pts, data, alpha)
137  class(sll_t_cubic_spline_interpolator_1d), intent(inout) :: this
138  !class(sll_t_cubic_spline_1d), intent(in) :: this
139  sll_int32, intent(in) :: num_pts
140  sll_real64, intent(in) :: alpha
141  sll_real64, dimension(num_pts), intent(inout) :: data
142 
143  ! local variables
144  sll_real64, dimension(num_pts) :: coordinates
145  sll_real64 :: length, delta
146  sll_real64 :: xmin, xmax
147  sll_int32 :: i
148  ! compute the interpolating spline coefficients
149  call sll_s_cubic_spline_1d_compute_interpolant( data, this%spline )
150  ! compute array of coordinates where interpolation is performed from displacement
151  length = this%interpolation_points(this%num_points) - &
152  this%interpolation_points(1)
153  delta = this%interpolation_points(2) - this%interpolation_points(1)
154  xmin = this%interpolation_points(1)
155  xmax = this%interpolation_points(this%num_points)
156  if (this%bc_type == sll_p_periodic) then
157  ! The case alpha = 0.0 is problematic. We need to further try to make
158  ! this computation in general m re efficient, minimize the use of modulo
159  ! and even explore a uniform grid representation...
160  if( alpha == 0.0_f64 ) then
161  coordinates(:) = this%interpolation_points(:)
162  else ! alpha != 0.0
163  do i = 1, num_pts
164  coordinates(i) = xmin + &
165  modulo(this%interpolation_points(i) - xmin + alpha, length)
166 !!$ write (*,'(a,z,f21.16,a,i,a,z,f21.16)') 'xmin = ', &
167 !!$ xmin, xmin, ' coordinates(',i,') = ', coordinates(i), &
168 !!$ coordinates(i)
169  sll_assert(coordinates(i) >= xmin)
170  sll_assert(coordinates(i) <= xmax)
171  end do
172  end if
173  else ! any other BC? better a case statement
174  if (alpha < 0 ) then
175  do i = 1, num_pts
176  coordinates(i) = max(this%interpolation_points(i) + alpha, xmin)
177  sll_assert((xmin <=coordinates(i)).and.(coordinates(i) <= xmax))
178  end do
179  else
180  do i = 1, num_pts
181  coordinates(i) = min(this%interpolation_points(i) + alpha, xmax)
182  sll_assert((xmin <=coordinates(i)).and.(coordinates(i) <= xmax))
183  end do
184  endif
185  end if
186  call sll_s_cubic_spline_1d_eval_array( coordinates, data, num_pts, &
187  this%spline )
188  end subroutine spline_interpolate1d_disp_inplace

350  interpolator%num_points = num_points
351  sll_allocate(interpolator%interpolation_points(num_points),ierr)

353  delta = (xmax - xmin) / (num_points - 1)

356  interpolator%interpolation_points(i-1) + delta

359  interpolator%bc_type = bc_type
360  if (present(slope_left).and.present(slope_right)) then
361  if (present(fast_algorithm)) then
362  call sll_s_cubic_spline_1d_init( &
363  interpolator%spline, &

369  fast_algorithm)
