131  function new_cartesian_topology_6d(top_collective, procs_per_dimension, periodic)
132  type(sll_t_cartesian_topology_6d), pointer :: new_cartesian_topology_6d
133  type(sll_t_collective_t), intent(in) :: top_collective
134  integer, parameter :: nd = 6
135  sll_int32, intent(in) :: procs_per_dimension(nd)
136  logical, intent(in) :: periodic(nd)
137  ! disallow reordering of MPI ranks with MPI_Cart_create()
138  logical, parameter :: reorder = .false.
139  sll_int32 :: i, ierr
140 
141  sll_allocate(new_cartesian_topology_6d, ierr)
142 
143  new_cartesian_topology_6d%procs = procs_per_dimension
144  new_cartesian_topology_6d%periodic = periodic
145 
146  ! create a cartesian process topology, return a new communicator
147  call mpi_cart_create(top_collective%comm, nd, &
148  new_cartesian_topology_6d%procs, &
149  new_cartesian_topology_6d%periodic, &
150  reorder, &
151  new_cartesian_topology_6d%comm, &
152  ierr)
153  sll_assert(ierr == mpi_success)
154 
155  ! query the coordinates of the current process within the cartesian topology
156  new_cartesian_topology_6d%coords = -1
157  call mpi_cart_get(new_cartesian_topology_6d%comm, nd, &
158  new_cartesian_topology_6d%procs, &
159  new_cartesian_topology_6d%periodic, &
160  new_cartesian_topology_6d%coords, &
161  ierr)
162  sll_assert(ierr == mpi_success)
163 
164  ! determine the neighbors within the cartesian topology
165  new_cartesian_topology_6d%neighbors = -1
166  do i = 1, nd
167  call mpi_cart_shift(new_cartesian_topology_6d%comm, i - 1, 1, &
168  new_cartesian_topology_6d%neighbors(2*i - 1), &
169  new_cartesian_topology_6d%neighbors(2*i), &
170  ierr)
171  sll_assert(ierr == mpi_success)
172  end do
173  end function

175  function new_cartesian_domain_decomposition_6d(topology, grid_size, halo_width)
176  type(sll_t_decomposition_6d), pointer :: new_cartesian_domain_decomposition_6d
177  type(sll_t_cartesian_topology_6d), pointer, intent(in) :: topology
178  integer, parameter :: nd = 6
179  sll_int32, intent(in) :: grid_size(nd)
180  sll_int32, intent(in) :: halo_width(nd)
181  sll_int32 :: i, ierr
182  sll_int32 :: lp, l0, l1
183 
184  ! --- initial checks
185  do i = 1, nd
186  sll_assert(halo_width(i) >= 0)
187  end do
188  do i = 1, nd
189  ! for the moment, the global grid must be divisible evenly among the MPI processes
190  sll_assert(mod(grid_size(i), topology%procs(i)) == 0)
191  end do
192 
193  ! --- copy and calculate all the necessary index values ---
194  sll_allocate(new_cartesian_domain_decomposition_6d, ierr)
195 
196  new_cartesian_domain_decomposition_6d%global = grid_size
197 
198  ! --- loop over dimensions and compute index values for each dimension
199  do i = 1, nd
200  ! compute the local number of grid points
201  lp = grid_size(i)/topology%procs(i)
202  ! compute the lower local index bound (the coords array starts at zero)
203  l0 = 1 + topology%coords(i)*lp
204  ! compute the upper local index bound (the coords array starts at zero)
205  l1 = (topology%coords(i) + 1)*lp
206 
207  sll_assert(lp/2 >= halo_width(i))
208 
209  new_cartesian_domain_decomposition_6d%local%mn(i) = l0
210  new_cartesian_domain_decomposition_6d%local%mx(i) = l1
211  new_cartesian_domain_decomposition_6d%local%hw(i) = halo_width(i)
212  new_cartesian_domain_decomposition_6d%local%lo(i) = l0 - halo_width(i)
213  new_cartesian_domain_decomposition_6d%local%hi(i) = l1 + halo_width(i)
214 
215  new_cartesian_domain_decomposition_6d%local%tx_lolo(i) = l0
216  new_cartesian_domain_decomposition_6d%local%tx_lohi(i) = l0 + (halo_width(i) - 1)
217  new_cartesian_domain_decomposition_6d%local%tx_hilo(i) = l1 - (halo_width(i) - 1)
218  new_cartesian_domain_decomposition_6d%local%tx_hihi(i) = l1
219  new_cartesian_domain_decomposition_6d%local%rx_lolo(i) = l0 - halo_width(i)
220  new_cartesian_domain_decomposition_6d%local%rx_lohi(i) = l0 - 1
221  new_cartesian_domain_decomposition_6d%local%rx_hilo(i) = l1 + 1
222  new_cartesian_domain_decomposition_6d%local%rx_hihi(i) = l1 + halo_width(i)
223  end do
224 
225  ! compute net array width
226  new_cartesian_domain_decomposition_6d%local%nw = 1 + &
227  new_cartesian_domain_decomposition_6d%local%mx - new_cartesian_domain_decomposition_6d%local%mn
228 
229  ! compute gross array width
230  new_cartesian_domain_decomposition_6d%local%gw = 1 + &
231  new_cartesian_domain_decomposition_6d%local%hi - new_cartesian_domain_decomposition_6d%local%lo
232  end function

287  subroutine apply_halo_exchange_6d_real64(topo, decomp, arr)
288  type(sll_t_cartesian_topology_6d), intent(in) :: topo
289  type(sll_t_decomposition_6d), intent(in) :: decomp
290  sll_real64, dimension(:, :, :, :, :, :), intent(inout) :: arr
291 
292  integer, save :: bufsize = 0
293  sll_real64, dimension(:), allocatable, save :: sendbuf, recvbuf
294  integer :: nxc
295 
296  integer, parameter :: nd = 6 ! we handle 6 dimensions
297  integer, dimension(:, :) :: ranges(nd, 2) ! index ranges for the copy operations
298 
299  integer :: id, jd
300  integer :: ierr
301 
302  ! --- loop over dimensions and exchange data between neighbors
303  do id = 1, nd
304  ! calculate the number of items to be exchanged
305  nxc = decomp%local%hw(id)
306  do jd = 1, nd
307  if (jd == id) continue
308  nxc = nxc*decomp%local%nw(jd)
309  end do
310 
311  ! allocate memory if necessary
312  if (nxc > bufsize) then
313  if (allocated(sendbuf)) deallocate (sendbuf)
314  if (allocated(recvbuf)) deallocate (recvbuf)
315  allocate (sendbuf(nxc))
316  allocate (recvbuf(nxc))
317  bufsize = nxc
318  end if
319 
320  ! --- (1) copy halo cells to the left neighbor
321  ranges(:, 1) = decomp%local%mn(:)
322  ranges(:, 2) = decomp%local%mx(:)
323  ranges(id, 1) = decomp%local%tx_lolo(id)
324  ranges(id, 2) = decomp%local%tx_lohi(id)
325  call copy_array_to_buffer_6d_real64(arr, decomp%local%lo, decomp%local%hi, sendbuf, ranges)
326 
327  call mpi_sendrecv(sendbuf, nxc, mpi_double_precision, topo%neighbors(2*id - 1), 1, &
328  recvbuf, nxc, mpi_double_precision, topo%neighbors(2*id), 1, &
329  topo%comm, mpi_status_ignore, ierr)
330 
331  ranges(:, 1) = decomp%local%mn(:)
332  ranges(:, 2) = decomp%local%mx(:)
333  ranges(id, 1) = decomp%local%rx_hilo(id)
334  ranges(id, 2) = decomp%local%rx_hihi(id)
335  call copy_buffer_to_array_6d_real64(recvbuf, arr, decomp%local%lo, decomp%local%hi, ranges)
336 
337  ! --- (2) copy halo cells to the right neighbor
338  ranges(:, 1) = decomp%local%mn(:)
339  ranges(:, 2) = decomp%local%mx(:)
340  ranges(id, 1) = decomp%local%tx_hilo(id)
341  ranges(id, 2) = decomp%local%tx_hihi(id)
342  call copy_array_to_buffer_6d_real64(arr, decomp%local%lo, decomp%local%hi, sendbuf, ranges)
343 
344  call mpi_sendrecv(sendbuf, nxc, mpi_double_precision, topo%neighbors(2*id), 1, &
345  recvbuf, nxc, mpi_double_precision, topo%neighbors(2*id - 1), 1, &
346  topo%comm, mpi_status_ignore, ierr)
347 
348  ranges(:, 1) = decomp%local%mn(:)
349  ranges(:, 2) = decomp%local%mx(:)
350  ranges(id, 1) = decomp%local%rx_lolo(id)
351  ranges(id, 2) = decomp%local%rx_lohi(id)
352  call copy_buffer_to_array_6d_real64(recvbuf, arr, decomp%local%lo, decomp%local%hi, ranges)

354  end do
