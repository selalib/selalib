172  subroutine solve_circulant(self, eigvals, rhs, res)
173  class(sll_t_maxwell_1d_fem) :: self
174  sll_real64, intent(in) :: eigvals(:) ! eigenvalues of circulant matrix
175  sll_real64, intent(in) :: rhs(:)
176  sll_real64, intent(out) :: res(:)
177  ! local variables
178  sll_int32 :: k
179  sll_real64 :: re, im
180 
181  ! Compute res from rhs, using eigenvalue of circulant matrix
182  res = rhs
183  ! Forward FFT
184  call sll_s_fft_exec_r2r_1d ( self%plan_fw, res, self%wsave )
185  self%wsave(1) = self%wsave(1) * eigvals(1)
186  do k=2, self%n_dofs/2
187  re = self%wsave(k) * eigvals(k) - &
188  self%wsave(self%n_dofs-k+2) * eigvals(self%n_dofs-k+2)
189  im = self%wsave(k) * eigvals(self%n_dofs-k+2) + &
190  self%wsave(self%n_dofs-k+2) * eigvals(k)
191  self%wsave(k) = re
192  self%wsave(self%n_dofs-k+2) = im
193  end do
194  self%wsave(self%n_dofs/2+1) = self%wsave(self%n_dofs/2+1)*eigvals(self%n_dofs/2+1)
195  ! Backward FFT
196  call sll_s_fft_exec_r2r_1d( self%plan_bw, self%wsave, res )
197  ! normalize
198  res = res / real(self%n_dofs, f64)
199  end subroutine solve_circulant

326  subroutine init_1d_fem( self, domain, n_dofs, s_deg_0 )
327  class(sll_t_maxwell_1d_fem), intent(out) :: self
328  sll_real64, intent(in) :: domain(2) ! xmin, xmax
329  sll_int32, intent(in) :: n_dofs ! number of degrees of freedom (here number of cells and grid points)
330  !sll_real64 :: delta_x ! cell size
331  sll_int32, intent(in) :: s_deg_0 ! highest spline degree
332 
333  ! local variables
334  sll_int32 :: ierr
335  sll_int32 :: j, k ! loop variables
336  sll_real64 :: coef0, coef1, sin_mode, cos_mode
337 
338  self%n_dofs = n_dofs
339  self%Lx = domain(2) - domain(1)
340  self%delta_x = self%Lx / real(n_dofs, f64)
341  self%s_deg_0 = s_deg_0
342  self%s_deg_1 = s_deg_0 - 1
343 
344  sll_allocate(self%mass_0(s_deg_0+1), ierr)
345  sll_allocate(self%mass_1(s_deg_0), ierr)
346 
347  select case(s_deg_0)
348  case(1) ! linear and constant splines
349  ! Upper diagonal coeficients of linear spline mass matrix (from Eulerian numbers)
350  self%mass_0(1) = 4.0_f64/6.0_f64
351  self%mass_0(2) = 1.0_f64/6.0_f64
352  ! Upper diagonal coeficients of constant spline mass matrix
353  self%mass_1(1) = 1.0_f64
354  case(2) ! quadratic and linear splines
355  ! Upper diagonal coeficients of quadratic spline mass matrix (from Eulerian numbers)
356  self%mass_0(1) = 66.0_f64/120.0_f64
357  self%mass_0(2) = 26.0_f64/120.0_f64
358  self%mass_0(3) = 1.0_f64/120.0_f64
359  ! Upper diagonal coeficients of linear spline mass matrix (from Eulerian numbers)
360  self%mass_1(1) = 4.0_f64/6.0_f64
361  self%mass_1(2) = 1.0_f64/6.0_f64
362  case(3)
363  ! Upper diagonal coeficients of cubic spline mass matrix (from Eulerian numbers)
364  self%mass_0(1) = 2416.0_f64/5040.0_f64
365  self%mass_0(2) = 1191.0_f64/5040.0_f64
366  self%mass_0(3) = 120.0_f64/5040.0_f64
367  self%mass_0(4) = 1.0_f64/5040.0_f64
368  ! Upper diagonal coeficients of quadratic spline mass matrix (from Eulerian numbers)
369  self%mass_1(1) = 66.0_f64/120.0_f64
370  self%mass_1(2) = 26.0_f64/120.0_f64
371  self%mass_1(3) = 1.0_f64/120.0_f64
372 
373  case default
374  print*, 'sll_t_maxwell_1d_fem init: spline degree ', s_deg_0, ' not implemented'
375  end select
376 
377  sll_allocate(self%eig_mass0(n_dofs), ierr)
378  sll_allocate(self%eig_mass1(n_dofs), ierr)
379  sll_allocate(self%eig_weak_ampere(n_dofs), ierr)
380  sll_allocate(self%eig_weak_poisson(n_dofs), ierr)
381  ! Initialise FFT
382  sll_allocate(self%work(n_dofs),ierr)
383  sll_allocate(self%wsave(n_dofs),ierr)
384  call sll_s_fft_init_r2r_1d( self%plan_fw, self%n_dofs, self%work, self%wsave, sll_p_fft_forward, normalized=.false. )
385  call sll_s_fft_init_r2r_1d( self%plan_bw, self%n_dofs, self%work, self%work, sll_p_fft_backward, normalized=.false. )
386 
387  ! Compute eigenvalues of circulant Ampere update matrix M_0^{-1} D^T M_1
388  ! and circulant Poisson Matrix (D^T M_1 D)^{-1}
389  ! zero mode vanishes due to derivative matrix D^T
390  self%eig_weak_ampere(1) = 0.0_f64
391  self%eig_weak_poisson(1) = 0.0_f64 ! Matrix is not invertible: 0-mode is set to 0
392  self%eig_mass0(1) = 1.0_f64 ! sum of coefficents is one
393  self%eig_mass1(1) = 1.0_f64 ! sum of coefficents is one
394 
395  do k=1, n_dofs/2 - 1
396  coef0 = self%mass_0(1)
397  coef1 = self%mass_1(1)
398  do j=1,s_deg_0 - 1
399  cos_mode = cos(2*sll_p_pi*j*k/real(n_dofs,f64))
400  coef0 = coef0 + 2* self%mass_0(j+1)*cos_mode
401  coef1 = coef1 + 2* self%mass_1(j+1)*cos_mode
402  enddo
403  ! add last term for larger matrix
404  j = s_deg_0
405  coef0 = coef0 + 2* self%mass_0(j+1)*cos(2*sll_p_pi*j*k/real(n_dofs,f64))
406  ! compute eigenvalues
407  self%eig_mass0(k+1) = coef0 ! real part
408  self%eig_mass0(n_dofs-k+1) = 0.0_f64 ! imaginary part
409  self%eig_mass1(k+1) = coef1 ! real part
410  self%eig_mass1(n_dofs-k+1) = 0.0_f64 ! imaginary part
411  cos_mode = cos(2*sll_p_pi*k/real(n_dofs,f64))
412  sin_mode = sin(2*sll_p_pi*k/real(n_dofs,f64))
413  self%eig_weak_ampere(k+1) = (coef1 / coef0) * (1-cos_mode) ! real part
414  self%eig_weak_ampere(n_dofs-k+1) = -(coef1 / coef0) * sin_mode ! imaginary part
415  self%eig_weak_poisson(k+1) = 1.0_f64 / (coef1 * ((1-cos_mode)**2 + &
416  sin_mode**2)) ! real part
417  self%eig_weak_poisson(n_dofs-k+1) = 0.0_f64 ! imaginary part
418  enddo
419  ! N/2 mode
420  coef0 = self%mass_0(1)
421  coef1 = self%mass_1(1)
422  do j=1, s_deg_0 - 1
423  coef0 = coef0 + 2 * self%mass_0(j+1)*cos(sll_p_pi*j)
424  coef1 = coef1 + 2 * self%mass_1(j+1)*cos(sll_p_pi*j)
425  enddo
426 
427  ! add last term for larger matrix
428  j = s_deg_0
429  coef0 = coef0 + 2 * self%mass_0(j+1)*cos(sll_p_pi*j)
430 
431  ! compute eigenvalues
432  self%eig_mass0(n_dofs/2+1) = coef0
433  self%eig_mass1(n_dofs/2+1) = coef1
434  self%eig_weak_ampere(n_dofs/2+1) = 2.0_f64 * (coef1 / coef0)
435  self%eig_weak_poisson(n_dofs/2+1) = 1.0_f64 / (coef1 *4.0_f64)
436 
437  end subroutine init_1d_fem
