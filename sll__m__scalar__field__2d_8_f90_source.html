491 sll_allocate(values(nptsx1,nptsx2),ierr)
492 
493 ! Fill the arrays with the needed information.
494 do j=1, nptsx2
495  eta2 = mesh%eta2_min + (j-1)*mesh%delta_eta2
496  do i=1, nptsx1
497  eta1 = mesh%eta1_min + (i-1)*mesh%delta_eta1
498  x1coords(i,j) = t%x1(eta1,eta2)
499  x2coords(i,j) = t%x2(eta1,eta2)
500  values(i,j) = field%value_at_point(eta1,eta2)
501  end do
502 end do

655 ! There is a bit of background history to the existence of the
656 ! free_data_discrete_2d() and reset_ptr_discrete_2d() routines. By request
657 ! of a user, the default behavior of the field is to manage its own copy
658 ! of the data. Thus upon creation, the object allocates the necessary
659 ! memory to store nodal values. However, it may be desired that the object
660 ! DOES NOT manage its own memory, but rather that it only points to some
661 ! external block of memory. To permit the change of the behavior of the
662 ! field from its default, is the role of these functions. The first
663 ! deallocates the memory and sets the flag which indicates that the field
664 ! no longer owns its memory. The second permits to reset the data pointer
665 ! to whatever is desired.

672  'data has been already freed or never allocated.'
