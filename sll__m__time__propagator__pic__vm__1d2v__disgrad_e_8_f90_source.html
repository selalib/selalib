301 
302  if(xnew < self%x_min .or. xnew > self%x_max )then

433  ! Accumulate jy

436 
437  ! Evaulate E_x at particle position and propagate v a half step
438  call self%kernel_smoother_1%evaluate &

440  ! Evaulate E_y at particle position and propagate v a half step

447  if (self%particle_group%group(i_sp)%n_weights == 3) then
448  ! Update weights if control variate
449  wp = self%particle_group%group(i_sp)%get_weights(i_part)
450  wp(3) = self%control_variate%cv(i_sp)%update_df_weight( xi(1:1), vi(1:2), 0.0_f64, wp(1), wp(2))
451  call self%particle_group%group(i_sp)%set_weights(i_part, wp)
452  end if

454  end do

462

471  call self%filter%apply_inplace( self%j_dofs(:,2) )
472

475  end if
476 
477  if( self%adiabatic_electrons ) then

480  call self%linear_op_schur_phiv%dot(self%phi_dofs, rhs0)
481  call self%maxwell_solver%multiply_gt( self%j_dofs(1:self%n_total1,1), self%j_dofs(:,2) )

483

486  call self%linear_solver_schur_phiv%set_guess(self%phi_dofs)
487  call self%linear_solver_schur_phiv%solve( rhs0, self%phi_dofs )
488  call self%maxwell_solver%multiply_g( self%phi_dofs, self%efield_dofs(1:self%n_total1,1) )
489  self%efield_dofs(:,1) = -self%efield_dofs(:,1)
490  else
491

494  call self%linear_operator_1%dot(self%efield_dofs(1:self%n_total1,1) , rhs1 )
495  rhs1 = rhs1 - dt * self%force_sign * self%j_dofs(1:self%n_total1,1)
496  ! Solve the Schur complement
497  self%linear_operator_1%sign = self%force_sign
498  call self%linear_solver_1%set_guess(self%efield_dofs(1:self%n_total1,1))
499  call self%linear_solver_1%solve( rhs1 , self%efield_dofs(1:self%n_total1,1) )
500 
501  ! Compute rhs
502  self%linear_operator_0%sign = -self%force_sign
503  call self%linear_operator_0%dot(self%efield_dofs(:,2) , rhs0 )
504  rhs0 = rhs0 - self%force_sign * dt * self%j_dofs(:,2)
505 
506  ! Solve the Schur complement
507  self%linear_operator_0%sign = self%force_sign
508  call self%linear_solver_0%set_guess(self%efield_dofs(:,2))
509  call self%linear_solver_0%solve( rhs0 , self%efield_dofs(:,2) )
510  end if
511

514  self%efield_dofs(self%n_total0,2) = 0._f64
515  end if

517  call self%filter%apply( self%efield_dofs(:,1), self%efield_filter(:,1) )
518  call self%filter%apply( self%efield_dofs(:,2), self%efield_filter(:,2) )
519  ! Second particle loop (second half step of particle propagation)
520  do i_sp=1,self%particle_group%n_species
521  qoverm = self%particle_group%group(i_sp)%species%q_over_m();
522  do i_part = 1, self%particle_group%group(i_sp)%n_particles
523 
524  vi = self%particle_group%group(i_sp)%get_v(i_part)
525  xi = self%particle_group%group(i_sp)%get_x(i_part)
526  ! Evaulate E_x at particle position and propagate v a half step
527  call self%kernel_smoother_1%evaluate &
528  (xi(1), self%efield_filter(1:self%n_total1,1), efield(1))
529  ! Evaulate E_y at particle position and propagate v a half step
530  call self%kernel_smoother_0%evaluate &
531  (xi(1), self%efield_filter(:,2), efield(2))
532 
533  vi(1:2) = vi(1:2) + dt* 0.5_f64* qoverm * efield
534  call self%particle_group%group(i_sp)%set_v( i_part, vi )
535 
536  if (self%particle_group%group(i_sp)%n_weights == 3) then
537  ! Update weights if control variate
538  wp = self%particle_group%group(i_sp)%get_weights(i_part)
539  wp(3) = self%control_variate%cv(i_sp)%update_df_weight( xi(1:1), vi(1:2), 0.0_f64, wp(1), wp(2))
540  call self%particle_group%group(i_sp)%set_weights(i_part, wp)
541  end if
542  end do
543  end do
