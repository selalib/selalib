93  subroutine solve_penta_diagonal(a, b, c, f, plan)
94 
95  sll_real64 :: a, b, c
96  sll_real64, dimension(:) :: f
97  type(sll_t_penta_diagonal_solver), pointer :: plan
98  sll_real64 :: s, t, p, l1, l2
99  sll_real64 :: d, d1, d2
100  sll_int32 :: n, i
101  sll_real64 :: sign_of_a=0.0_f64
102 
103  if ( abs(a) <= 2.0_f64*(abs(b)+abs(c)) ) then
104  print*, 'a, b, and c must be such that: |a| > 2(|b|+|c|)'
105  print*, a, b, c
106  print*, 'Exiting...'
107  stop
108  endif
109 
110  n = plan%n
111 
112  s = (a/2+c)*(a/2+c) - b*b
113  t = a*a/2 - b*b - 2*c*c
114 
115  if (a/=0.0_f64) then
116  sign_of_a = a/abs(a)
117  endif
118 
119  p = (a-2*c)/4 + sign_of_a*sqrt(sign_of_a*(a-2*c)* &
120  sqrt(s)+t)/2.0_f64 + sign_of_a*sqrt(s)/2.0_f64
121  l1 = b/(p+c)
122  l2 = c/p
123 
124  do i=1,n
125  plan%y(i) = f(i)/p
126  plan%e1(i) = 0.0_f64
127  plan%e2(i) = 0.0_f64
128  enddo
129  plan%e1(1) = 1.0_f64
130  plan%e2(2) = 1.0_f64
131 
132  plan%y = solve_subsystem(l1, l2, plan%y, n)
133  plan%z = solve_subsystem(l1, l2, plan%e1, n)
134  plan%w = solve_subsystem(l1, l2, plan%e2, n)
135 
136  d = 1.0_f64 + l1*l2*(plan%z(2)+plan%w(1)) + l2*l2*(plan%z(1)+plan%w(2)) + &
137  l1*l1*plan%z(1) + l2**4*(plan%z(1)*plan%w(2)-plan%z(2)*plan%w(1))
138 
139  d1 = l2**4*(plan%y(1)*plan%w(2)-plan%y(2)*plan%w(1)) + &
140  (l1*l1+l2*l2)*plan%y(1) + l1*l2*plan%y(2)
141 
142  d2 = l2**4*(plan%y(2)*plan%z(1)-plan%y(1)*plan%z(2)) + &
143  l1*l2*plan%y(1) + l2*l2*plan%y(2)
144 
145  do i=1,n
146  plan%solution(i) = plan%y(i) - (d1*plan%z(i)+d2*plan%w(i))/d
147  enddo
148 
149  end subroutine solve_penta_diagonal

152  function solve_subsystem(l1, l2, b, n) result (x)
153 
154  sll_real64 :: l1, l2
155  sll_real64, dimension(:) :: b
156  sll_int32 :: n, i
157  sll_real64, dimension(n) :: x, y
158  sll_real64 :: tmp
159 
160  y(1) = b(1)
161  y(2) = b(2) - l1*y(1)
162 
163  do i=3,n
164  tmp=b(i) - ( l2*y(i-2) + l1*y(i-1) )
165  if(abs(tmp)<1e-30_f64)then

168  y(i) = tmp

170

175  if(abs(tmp)<1e-30_f64)then
