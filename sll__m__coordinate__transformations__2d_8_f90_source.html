867  subroutine sll_s_coordinate_transformation_2d_discrete_init( &
868  transf, &
869  mesh_2d, &
870  label, &
871  x1_interpolator, &
872  x2_interpolator, &
873  jacobians_n_interpolator, &
874  x1_node, &
875  x2_node, &
876  jacobians_node, &
877  x1_cell, &
878  x2_cell, &
879  jacobians_cell )
880 
881  class(sll_t_coordinate_transformation_2d_discrete) :: transf
882  type(sll_t_cartesian_mesh_2d), target :: mesh_2d
883  character(len=*), intent(in) :: label
884 
885  class(sll_c_interpolator_2d), target :: x1_interpolator
886  class(sll_c_interpolator_2d), target :: x2_interpolator
887  class(sll_c_interpolator_2d), target :: jacobians_n_interpolator
888  sll_real64, dimension(:,:), intent(in), optional :: x1_node
889  sll_real64, dimension(:,:), intent(in), optional :: x2_node
890  sll_real64, dimension(:,:), intent(in), optional :: jacobians_node
891  sll_real64, dimension(:,:), intent(in), optional :: jacobians_cell
892  sll_real64, dimension(:,:), intent(in), optional :: x1_cell
893  sll_real64, dimension(:,:), intent(in), optional :: x2_cell
894 
895  sll_real64 :: eta_1
896  sll_real64 :: eta_2
897  sll_real64 :: eta_1_min
898  sll_real64 :: eta_2_min
899  sll_real64 :: delta_eta_1
900  sll_real64 :: delta_eta_2
901  sll_real64 :: jacobian_val
902  sll_int32 :: i
903  sll_int32 :: j
904  sll_int32 :: ierr
905  sll_int32 :: npts1
906  sll_int32 :: npts2
907  logical :: x1n
908  logical :: x2n
909  logical :: x1c
910  logical :: x2c
911  logical :: jc
912  logical :: jn
913 
914  transf%mesh => mesh_2d
915  transf%label = trim(label)
916  x1n = present(x1_node)
917  x2n = present(x2_node)
918  x1c = present(x1_cell)
919  x2c = present(x2_cell)
920  jc = present(jacobians_cell)
921  jn = present(jacobians_node)
922  npts1 = mesh_2d%num_cells1 + 1
923  npts2 = mesh_2d%num_cells2 + 1
924 
925  ! Check argument consistency
926  ! DISCRETE_MAPs require only some of the parameters. If the mapping is
927  ! defined from the nodes of the logical (eta1, eta2) mesh to the nodes
928  ! of the physical mesh (x1,x2), then either:
929  ! - the node arrays are required: jacobians_node, x1_node and x2_node. Or
930  ! - the x1_interpolator and x2_interpolator must contain already the
931  ! coefficient information that would permit the calculation of the
932  ! x1 and x2 points.
933  !
934  ! If the transformation is done on the points at the center of the cells
935  ! then these parameters are also required:
936  ! jacobians_cell, x1_cell, x2_cell.
937  ! node and cell values are not mutually exclusive, thus all 6 parameters
938  ! can be provided in the discrete case. It is up to the caller to make
939  ! sure that the data set is consistent.
940 
941  ! 1. Check that the discrete representation of the transformation is
942  ! consistent with the size of the 2D array.
943 
944  if( ( x1n .and. (.not.x2n) ) .or. &
945  ((.not.x1n) .and. x2n) ) then
946 
947  print *,'ERROR, sll_s_coordinate_transformation_2d_discrete_init():', &
948  'for the moment, this function does not support specifying ', &
949  'transformation only with one of the node arrays x1_node or ', &
950  'x2_node. Either pass both, or none, but with the ', &
951  'corresponding interpolators having their coefficients ', &
952  'already set.'
953  stop
954  call jacobians_n_interpolator%delete() !PN added to remove the warning
955  end if
956 
957  if( x1n ) then
958  if( size(x1_node,1) .lt. npts1 ) then
959  print *,'ERROR, sll_s_coordinate_transformation_2d_discrete_init()',&
960  ' the size of the x1_node arrays is ', &
961  'inconsistent with the number of points declared, ', &
962  'in the logical mesh.'
963  stop
964  end if
965  end if
966 
967  if( x2n ) then
968  if( size(x1_node,2) .lt. npts2 ) then
969  print *, 'ERROR, sll_s_coordinate_transformation_2d_discrete_init()',&
970  ' the size of the x2_node arrays is ', &
971  'inconsistent with the number of points declared, ', &
972  'in the logical mesh.'
973  stop
974  end if
975  end if
976 
977  if( jn .eqv. .true. ) then
978  if( &
979  (size(jacobians_node,1) .lt. npts1 - 1 ) .or. &
980  (size(jacobians_node,2) .lt. npts2 - 1 ) ) then
981  print *, 'ERROR, sll_s_coordinate_transformation_2d_discrete_init()',&
982  ': the size of the jacobians_node array is ', &
983  'inconsistent with the number of points declared, ', &
984  'npts1 or npts2.'
985  stop
986  end if
987  end if
988 
989  if( jc .eqv. .true. ) then
990  if( &
991  (size(jacobians_cell,1) .lt. npts1 - 1 ) .or. &
992  (size(jacobians_cell,2) .lt. npts2 - 1 ) ) then
993  print *, 'ERROR, sll_s_coordinate_transformation_2d_discrete_init()',&
994  ': the size of the jacobians_cell arrays is ', &
995  'inconsistent with the number of points declared, ', &
996  'npts1 or npts2.'
997  stop
998  end if
999  end if
1000 
1001  transf%x1_interp => x1_interpolator
1002  transf%x2_interp => x2_interpolator
1003 
1004  ! Allocate the arrays for precomputed jacobians.
1005  sll_allocate(transf%jacobians_n(npts1,npts2), ierr)
1006  sll_allocate(transf%jacobians_c(npts1-1, npts2-1), ierr)
1007 
1008  ! Allocation for x1 and x2 at nodes
1009  sll_allocate(transf%x1_node(npts1,npts2), ierr)
1010  sll_allocate(transf%x2_node(npts1,npts2), ierr)
1011 
1012  ! Allocation for x1 and x2 at cells
1013  sll_allocate(transf%x1_cell(npts1-1,npts2-1), ierr)
1014  sll_allocate(transf%x2_cell(npts1-1,npts2-1), ierr)
1015 
1016  ! initialize the local arrays. Note that since the map has its
1017  ! own copies, it owns this information locally and will destroy
1018  ! this information when the object is deleted. The caller is
1019  ! thus responsible for deallocating the arrays that were passed as
1020  ! arguments.
1021 
1022  eta_1_min = mesh_2d%eta1_min
1023  eta_2_min = mesh_2d%eta2_min
1024  delta_eta_1 = mesh_2d%delta_eta1
1025  delta_eta_2 = mesh_2d%delta_eta2
1026 
1027  if( x1n .and. x2n ) then
1028  do j=1, npts2
1029  do i=1, npts1
1030  transf%x1_node(i,j) = x1_node(i,j)
1031  transf%x2_node(i,j) = x2_node(i,j)
1032  end do
1033  end do
1034  else
1035  if(x1_interpolator%coefficients_are_set() .eqv. .false.) then
1036  print *, 'ERROR, sll_s_coordinate_transformation_2d_discrete_init()',&
1037  ': the x1_node array was not passed and the corresponding ', &
1038  'interpolator has no initialized coefficients. Exiting...'
1039  stop
1040  endif
1041  if(x2_interpolator%coefficients_are_set() .eqv. .false.) then
1042  print *, 'ERROR, sll_s_coordinate_transformation_2d_discrete_init()',&
1043  ': the x2_node array was not passed and the corresponding ', &
1044  'interpolator has no initialized coefficients. Exiting...'
1045  stop
1046  endif
1047  ! now initialize the arrays starting from the interpolator information
1048  ! and the logical mesh information.
1049  do j=0, npts2 - 1
1050  eta_2 = eta_2_min + real(j,f64)*delta_eta_2
1051  do i=0, npts1 - 1
1052  eta_1 = eta_1_min + real(i,f64)*delta_eta_1
1053  transf%x1_node(i+1,j+1) = &
1054  x1_interpolator%interpolate_from_interpolant_value(eta_1,eta_2)
1055  transf%x2_node(i+1,j+1) = &
1056  x2_interpolator%interpolate_from_interpolant_value(eta_1,eta_2)
1057  end do
1058  end do
1059  end if
1060 
1061  ! Compute the spline coefficients
1062  if( x1n .and. (x1_interpolator%coefficients_are_set() .eqv. .false.) ) then
1063  call x1_interpolator%compute_interpolants( transf%x1_node )
1064  end if
1065  if( x2n .and. (x2_interpolator%coefficients_are_set() .eqv. .false.) ) then
1066  call x2_interpolator%compute_interpolants( transf%x2_node )
1067  end if
